Table users {
  logged boolean [not null, default: false]
  user_id integer [not null,unique]
  username varchar [primary key,unique]
  premission_level varchar [not null, default: "basic_user"]
  cart text 
  created_at timestamp
}

Table computer_store {
  logged boolean [not null, default: false]
  user_id integer [not null, unique]
  premission_level varchar [not null, default: "basic_user"]

  product_id integer [unique, not null] 
  product_name varchar [primary key]
  description text 
  status varchar [not null, default: "in Stock"]
  created_at timestamp [not null]
}

Ref premissions: users.premission_level - computer_store.premission_level
